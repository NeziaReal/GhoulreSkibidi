if not game:IsLoaded() then
    game.Loaded:Wait()
end

local MAINMENU_PLACEID = 72368345999613
local MAIN_GAMEID = 6490954291
local GH_GAMEID = 9509842868
local TS_GAMEID = 4871329703
local BIZZARELINEAGE_GAMEID = 5130394318
local DISCORD_INVITE = "https://discord.gg/uu57TArNU6"

local qot = (syn and syn.queue_on_teleport) or queue_on_teleport or (fluxus and fluxus.queue_on_teleport)
if qot then
    qot([[loadstring(game:HttpGet("https://raw.githubusercontent.com/NeziaReal/GhoulreSkibidi/refs/heads/main/Nameless.txt"))()]])
end

local Notifications = loadstring(game:HttpGet("https://raw.githubusercontent.com/tacolift/notification-library/main/library.lua"))()


local MAIN_SCRIPT = loadstring(game:HttpGet("https://raw.githubusercontent.com/NeziaReal/GhoulreSkibidi/refs/heads/main/GhoulRe.lua"))
local Webhook_Script = loadstring(game:HttpGet("https://raw.githubusercontent.com/NeziaReal/jjs/refs/heads/main/extrav4"))
local MAIN_MENU_SCRIPT = loadstring(game:HttpGet("https://raw.githubusercontent.com/NeziaReal/GhoulreSkibidi/refs/heads/main/GameTeleport.lua"))
local TS_SCRIPT = loadstring(game:HttpGet("https://vss.pandadevelopment.net/virtual/file/c8e6386eed2a4a56"))()
local UNIVERSAL_SCRIPT = loadstring(game:HttpGet("https://raw.githubusercontent.com/NeziaReal/GhoulreSkibidi/refs/heads/main/Universal.lua"))
local GH_SCRIPT = loadstring(game:HttpGet("https://raw.githubusercontent.com/NeziaReal/GhoulreSkibidi/refs/heads/main/GardenHoriz"))
local BIZZARELINEAGE_SCRIPT = loadstring(game:HttpGet("https://raw.githubusercontent.com/NeziaReal/GhoulreSkibidi/refs/heads/main/bl.lua"))

if game.PlaceId == MAINMENU_PLACEID then
    MAIN_MENU_SCRIPT() 
    task.wait(3)
    Webhook_Script()
    Notifications.Create("Nameless ⭐", "Loaded Main Menu Script", 2, Color3.fromRGB(255,255,255))
elseif game.GameId == MAIN_GAMEID then
    MAIN_SCRIPT() 
    task.wait(3)
    Webhook_Script()
    Notifications.Create("Nameless ⭐", "Loaded Ghoul Re Script", 2, Color3.fromRGB(255,255,255))
elseif game.GameId == GH_GAMEID then
    GH_SCRIPT()
    task.wait(3)
    Webhook_Script()
    Notifications.Create("Nameless ⭐", "Loaded Garden Horizon Script.", 2, Color3.fromRGB(255,255,255))
elseif game.GameId == TS_GAMEID then
    TS_SCRIPT()
    task.wait(3)
    Webhook_Script()
    Notifications.Create("Nameless ⭐", "Loaded Type Soul Script", 2, Color3.fromRGB(255,255,255))
elseif game.GameId == BIZZARELINEAGE_GAMEID then
    BIZZARELINEAGE_SCRIPT()
    task.wait(3)
    Webhook_Script()
    Notifications.Create("Nameless ⭐", "Loaded Bizzare Lineage Script", 2, Color3.fromRGB(255,255,255))
else
    pcall(function()
        setclipboard(DISCORD_INVITE)
    end)
    Notifications.Create("Nameless ⭐", "Nameless does not support this game.", 2, Color3.fromRGB(255,255,255))
    Notifications.Create("Nameless ⭐", "Loaded Universal Script", 2, Color3.fromRGB(255,255,255))
    UNIVERSAL_SCRIPT()
    Webhook_Script() 
end
